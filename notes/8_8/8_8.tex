\section*{Main Ideas}

\begin{itemize}

\item
Improper integrals are definite integrals with infinities or asymptotes within their bounds.

\item
This can be dealt with by replacing the infinity or asymptote with a limit.

\item
If the limit does not converge, the integral does not exist.

\end{itemize}

\section*{Problems}

\begin{itemize}

\item[1.]
\begin{flalign*}
&
\int_0^{\infty} \frac{dx}{x^2+1}
&&
\\
\\
&
=\lim_{b \to \infty}
\int_0^b \frac{dx}{x^2+1}
&&
\text{\color{blue} step 1: use a limit to remove infinity}
\\
\\
&
=\lim_{b \to \infty}
\left. \arctan x \right|_0^b
&&
\text{\color{blue} step 2: evaluate the integral}
\\
\\
&
=\lim_{b \to \infty}
\arctan b = \frac{\pi}{2}
&&
\text{\color{blue} step 3: evaluate bounds and limit}
\end{flalign*}

\item[3.]
\begin{flalign*}
&
\int_0^1 \frac{dx}{\sqrt{x}}
&&
\text{\color{blue} this integral has an asymptote}
\\
\\
&
=\lim_{t \to 0}
\int_t^1 \frac{dx}{\sqrt{x}}
&&
\text{\color{blue} step 1: place a limit at the asymptote}
\\
\\
&
=\lim_{t \to 0}
\left. 2 \sqrt{x} \right|_t^1
&&
\text{\color{blue} step 2: evaluate the integral}
\\
\\
&
=\lim_{t \to 0}
2 - 2 \sqrt{t} = 2 - 0 = 2
&&
\text{\color{blue} step 3: evaluate bounds and limit}
\end{flalign*}

\item[5.]
\begin{flalign*}
&
\int_{-1}^1 \frac{dx}{x^{2/3}}
&&
\text{\color{blue} the issue here is the asymptote in the middle}
\\
\\
&
\int_{-1}^0 \frac{dx}{x^{2/3}}
+\int_0^1 \frac{dx}{x^{2/3}}
&&
\text{\color{blue} step 1: split integral into parts}
\\
\\
&
=\lim_{t \to 0^-}
\int_{-1}^t \frac{dx}{x^{2/3}}
+\lim_{t \to 0^+}
\int_t^1 \frac{dx}{x^{2/3}}
&&
\text{\color{blue} step 2: place limit at asymptote}
\\
\\
&
=\lim_{t \to 0^-}
\left. 3x^{1/3} \right|_{-1}^t
+\lim_{t \to 0^+}
\left. 3x^{1/3} \right|_t^1
&&
\text{\color{blue} step 3: evaluate integral}
\\
\\
&
=\lim_{t \to 0^-}
3t^{1/3} + - (-3)
+\lim_{t \to 0^+}
3 - 3t^{1/3}
=6
&&
\text{\color{blue} step 4: evaluate bounds and limit}
\end{flalign*}

\end{itemize}
