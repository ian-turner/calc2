\\
\\
=2+\frac{1}{10^n}
&
&
\text{\color{blue} re-write sequence}
\\
\\
\lim_{n \to \infty}
2+\frac{1}{10^n}

\\

&
&
\text{\color{blue} evaluate limit}
\end{align*}
